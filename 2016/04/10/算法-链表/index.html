单链表整体逆置/相邻元素逆置

/**

  • 单链表整体逆置 && 单链表相邻元素逆置
    */
    public class LinkList {
    static class Node{

    public int data;
public Node next;
public Node(int data){
    this.data = data;
}
public Node(){}

    }
    public static void main(String[] args) {

    //带头结点的链表
Node head = new Node();
head.next = null;
Node p = head;
for(int i = 1;i<10;i++){
    Node temp = new Node(i);
    temp.next = null;
    p.next = temp;
    p = p.next;
}
//单链表整体逆置
reverseLinkedList(head);
printLinkedList(head);

//单链表相邻元素逆置
neighbourReverse(head);
printLinkedList(head);

    }
    //反转相邻元素，交换相邻元素的值，然后令next指向后两个节点，若有一个不存在则结束
    private static void neighbourReverse(Node head) {

    int temp;
Node p=head.next;
Node q=p.next;
while(p!=null && q!=null){
    temp = p.data;
    p.data = q.data;
    q.data = temp;
    p = q.next;
    q = p.next;
}

    }

    //反转单链表
    private static Node reverseLinkedList(Node head) {

    Node p,q,r;
p = head.next;
q = p.next;
p.next = null;
while(q!=null){
     r=q.next;
     q.next=head.next;
     head.next=q;
     q=r;
}
return head;

    }

    // 反转链表(没有头结点)
    // 将结点保存至pNext，然后将当前结点的next重新设置为prev，将prev指向当前结点，再将指针后移一个结点
    private static Node reverseLink(Node head) {

    Node prev = null;
Node newHead = null;
Node p = head;
while(p!=null){
    Node pNext = p.next;
    if(pNext==null)
        newHead = p;
    p.next = prev;
    prev = p;
    p = pNext;
}
return newHead;

    }

    // 递归反转链表(没有头结点)
    private static Node reverseRecursively(Node head){

    if(head == null || head.next == null)
    return head;
else{
    Node newHead = reverseRecursively(head.next);
    head.next.next = head;
    head.next = null;
    return newHead;
}

    }
    //顺序打印输出(带头结点的链表输出)
    private static void printLinkedList(Node p) {

    p = p.next;
while(p!=null){
    System.out.println(p.data);
    p = p.next;
}

    }
    // 逆序打印结点(不带头结点)
    private static void printReverse(Node head) {

    Node p = head;
if(p!=null){
    if(p.next!=null)
        printReverse(p.next);
}
System.out.print(p.value+" ");

    }
    }
    ```
